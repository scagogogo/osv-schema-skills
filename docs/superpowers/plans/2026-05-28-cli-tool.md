# OSV Schema CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: `superpowers:subagent-driven-development`
> Steps use checkbox (`- [ ]`) syntax.

**Goal:** 将 osv-schema 库封装为 CLI 工具，支持解析、校验、格式化 OSV JSON 文件

**Architecture:** 用户通过命令行传入 OSV JSON 文件路径 → CLI 读取文件 → 调用 osv_schema 包的 UnmarshalFromJsonFile 反序列化 → 根据子命令执行不同操作（parse 展示关键字段、validate 校验格式、version 显示版本信息）→ 输出结果到 stdout。选择 cobra 作为 CLI 框架，因为它是 Go 生态最成熟的 CLI 库，与项目已有的 mapstructure/spf13 生态一致。

**Tech Stack:** Go 1.18 (兼容现有 go.mod), cobra v1.10.2, 现有 osv_schema 包

**Scope:** Small
**Risk:** Low

**Risks:**
- 添加 cobra 依赖会增加 go.sum 体积，但 cobra 是 Go 标准 CLI 库，风险极低
- 所有新代码在 cmd/ 目录下，不修改现有库代码，对下游用户零影响

**Autonomy Level:** Full

// cmd/osv/parse.go
package main

import (
	"encoding/json"
	"fmt"
	"os"
	"strings"

	osv_schema "github.com/scagogogo/osv-schema"
	"github.com/spf13/cobra"
)

var parseCmd = &cobra.Command{
	Use:   "parse <file>",
	Short: "Parse an OSV JSON file and display key fields",
	Long:  "Parse an OSV JSON file and display its vulnerability ID, summary, severity, affected packages and other key information.",
	Args:  cobra.ExactArgs(1),
	RunE:  runParse,
}

func init() {
	rootCmd.AddCommand(parseCmd)
}

func runParse(cmd *cobra.Command, args []string) error {
	filePath := args[0]
	osvData, err := parseOsvFile(filePath)
	if err != nil {
		return fmt.Errorf("failed to parse OSV file %s: %w", filePath, err)
	}

	if outputFormat == "json" {
		encoder := json.NewEncoder(os.Stdout)
		encoder.SetIndent("", "  ")
		return encoder.Encode(osvData)
	}
	return printParseText(osvData)
}

func printParseText(o *osv_schema.OsvSchema[any, any]) error {
	fmt.Printf("ID:             %s\n", o.ID)
	fmt.Printf("Schema Version: %s\n", o.SchemaVersion)
	fmt.Printf("Summary:        %s\n", o.Summary)

	if len(o.Aliases) > 0 {
		fmt.Printf("Aliases:        %s\n", strings.Join(o.Aliases, ", "))
		if cve := o.Aliases.GetCVE(); cve != "" {
			fmt.Printf("CVE:            %s\n", cve)
		}
	}

	if len(o.Severity) > 0 {
		fmt.Println("\nSeverity:")
		for _, s := range o.Severity {
			fmt.Printf("  %s: %s (score: %.1f)\n", s.Type, s.Score, s.GetScore())
		}
	}

	if len(o.Affected) > 0 {
		fmt.Println("\nAffected Packages:")
		for _, a := range o.Affected {
			if a.Package != nil {
				fmt.Printf("  %s/%s", a.Package.Ecosystem, a.Package.Name)
				if len(a.Versions) > 0 {
					fmt.Printf(" (versions: %s)", strings.Join(a.Versions, ", "))
				}
				fmt.Println()
			}
		}
	}

	if len(o.References) > 0 {
		fmt.Println("\nReferences:")
		for _, r := range o.References {
			fmt.Printf("  [%s] %s\n", r.Type, r.URL)
		}
	}

	return nil
}

```go
// cmd/osv/validate.go
package main

import (
	"encoding/json"
	"fmt"
	"os"

	osv_schema "github.com/scagogogo/osv-schema"
	"github.com/spf13/cobra"
)

type ValidationResult struct {
	File    string   `json:"file"`
	Valid   bool     `json:"valid"`
	Errors  []string `json:"errors,omitempty"`
	ID      string   `json:"id,omitempty"`
	Version string   `json:"schema_version,omitempty"`
}

var validateCmd = &cobra.Command{
	Use:   "validate <file> [file...]",
	Short: "Validate OSV JSON files against the schema",
	Long:  "Validate one or more OSV JSON files to check if they can be parsed and contain required fields (id, schema_version).",
	Args:  cobra.MinimumNArgs(1),
	RunE:  runValidate,
}

func init() {
	rootCmd.AddCommand(validateCmd)
}

func runValidate(cmd *cobra.Command, args []string) error {
	results := make([]*ValidationResult, 0, len(args))
	hasError := false

	for _, filePath := range args {
		result := validateFile(filePath)
		results = append(results, result)
		if !result.Valid {
			hasError = true
		}
	}

	if outputFormat == "json" {
		encoder := json.NewEncoder(os.Stdout)
		encoder.SetIndent("", "  ")
		_ = encoder.Encode(results)
	} else {
		for _, r := range results {
			if r.Valid {
				fmt.Printf("✓ %s (id=%s, schema_version=%s)\n", r.File, r.ID, r.Version)
			} else {
				fmt.Printf("✗ %s\n", r.File)
				for _, e := range r.Errors {
					fmt.Printf("  - %s\n", e)
				}
			}
		}
	}

	if hasError {
		os.Exit(1)
	}
	return nil
}

func validateFile(filePath string) *ValidationResult {
	result := &ValidationResult{File: filePath}

	// 检查文件是否存在
	raw, err := os.ReadFile(filePath)
	if err != nil {
		result.Errors = append(result.Errors, fmt.Sprintf("cannot read file: %v", err))
		return result
	}

	// 检查是否为合法 JSON
	if !json.Valid(raw) {
		result.Errors = append(result.Errors, "file is not valid JSON")
		return result
	}

	// 尝试解析为 OSV 结构
	osvData, err := osv_schema.UnmarshalFromJson[any, any](raw)
	if err != nil {
		result.Errors = append(result.Errors, fmt.Sprintf("OSV parse error: %v", err))
		return result
	}

	// 校验必要字段
	if osvData.ID == "" {
		result.Errors = append(result.Errors, "missing required field: id")
	}
	if osvData.SchemaVersion == "" {
		result.Errors = append(result.Errors, "missing required field: schema_version")
	}

	if len(result.Errors) == 0 {
		result.Valid = true
		result.ID = osvData.ID
		result.Version = osvData.SchemaVersion
	}
	return result
}
```
